//Many to one 
#include <mpi.h>
#include <stdio.h>

int main()
{
    int np;
    int pid;

    MPI_Init(NULL, NULL);
    MPI_Comm_size(MPI_COMM_WORLD, &np);
    MPI_Comm_rank(MPI_COMM_WORLD, &pid);

    if (pid == 0)
    {
        int received_num;
        MPI_Status status;

        for (int i = 1; i < np; i++) {
            MPI_Recv(&received_num, 1, MPI_INT, i, 50, MPI_COMM_WORLD, &status);
            printf("Process 0 received %d from Process %d\n", received_num, i);
        }
    }
    else
    {
        int num = pid * 10;  // each process sends a different value
        MPI_Send(&num, 1, MPI_INT, 0, 50, MPI_COMM_WORLD);
        printf("Process %d sent %d to Process 0\n", pid, num);
    }

    MPI_Finalize();
    return 0;
}
